##	Terraform - Security Group (r6i.4xlarge EC2)
---
provider "aws" {
	region	=	"us-east-1"
}

resource "aws_security_group" "pathnexsg" {
	name	=	"pathnexsg"
	description =	"create allow sg for pathnex"

	ingress {
		from_port	=	22
		to_port		=	22
		protocol	=	"tcp"
		cidr_blocks	=	["0.0.0.0/0"]
	}
}

resource "aws_instance" "pathnexEC2" {
	ami	=	"ami-id"
	instance_type	=	"r6i.4xlarge"
	vpc_security_group_ids	=	[aws_security_group.pathnexsg.id]

	tags = {
		Name	=	"pathnexEC2"
	}
}

##	NodeJS Build
#	Jenkins Pipeline - NodeJS Build

pipeline {
	agent any
	tools {
		nodejs 'NodeJS-16'
	}
	stages {
		stage ('Git-Checkout') {
			steps {
				git branch: 'main', url: '<git repo url>'
			}
		}
		stage ('Install NPM dependencies') {
			steps {
				sh 'npm install'
			}
		}
		stage ('Test-package') {
			steps {
				sh 'npm test'
			}
		}
		stage ('Build-package') {
			steps {
				sh 'npm run build'
			}
		}
	}
}
