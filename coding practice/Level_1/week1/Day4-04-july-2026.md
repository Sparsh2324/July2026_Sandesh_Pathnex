## Ansible task - Install multiple package

---
-   name:   Install multiple package
    hosts:  all
    become: yes

    tasks:
        -   name:   Install multiple package
            yum:
                name:
                    -   git
                    -   wget
                state:  present

##  Terraform task - EC2 Key pair (c6i.8xlarge)

---

provider "aws" {
    region  =   "us-east-1"
}

resource "aws_key_pair" "pathnex_key" {
    key_name    =   "pathnex_key"
    public_key  =   "<key_value>"
}

resource "aws_instance" "pathnexEC2" {
    ami =   "<ami_id>"
    instance_type   =   "c6i.8xlarge"
    key_name    =   aws_key_pair.pathnex_key.key_name

    tags    =   {
        Name    =   pathnexEC2
    }
}

## Jenkins Task - Parallel pipeline

pipeline {
    agent any

    stages {
        stage ('parallel task') {
            parallel {
                stage ('Build') {
                    steps {
                        sh 'mvn clean compile'
                    }
                }
                stage ('Test') {
                    steps {
                        sh 'mvn test"
                    }
                }
            }
        }
    }
}

## GitLab CI - Parallel build-test job

---

stages:
    -   build-test

maven-build:
    stage:  build-test
    image:  maven:3.8.1-jdk-17
    script:
        -   git clone "<git_repo>"
        -   cd "<repo_path>"
        -   mvn clean compile
    parallel:   1

maven-test:
    stage:  build-test
    image:  maven:3.8.1-jdk-17
    script:
        -   git clone "<git_repo>"
        -   cd  "<repo_path>"
        -   mvn test
    parallel:   1
