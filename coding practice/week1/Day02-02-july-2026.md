## Ansible Task - Install & Enable Nginx

-   name: Install & Enable Nginx
    hosts: all
    become: yes

    tasks:
        -   name: Install Nginx
            yum:
                name: nginx
                state: present

        -   name: Enable Nginx Service
            service:
                name: nginx
                state: started
                enabled: yes

## Kubernetes - Deployment with 2 replicas

apiVersion: v1
kind:   Deployment
metadata:
    name:   pathnex-deployment

spec:
    replicas: 2
    selector:
        matchLabels:
            app: pathnex-app
    template:
        metadata:
            labels:
                app: pathnex-app
        spec:
            containers:
                -   name:   app
                    image:  nginx
                    ports:
                        -   containerPort: 80

## Jenkins task - maven build
# you will learn how to **compile, test, and package a java project** using maven

pipeline {
    agent any
    tools {
        maven   'maven-3.8.1'
        jdk     'JDK-17'
    }
    stages {
        stage ("git_checkout") {
            steps {
                git branch: 'main', url: "git_repo_url"
            }
        }
        stage ("Code_compile") {
            steps {
                sh 'mvn clean compile'
            }
        }
        stage ("code_test") {
            steps {
                sh 'mvn test'
            }
        }
        stage ("code_package") {
            steps {
                sh 'mvn package'
            }
        }
    }
}

## GitLab CI - Build, Test & package using maven

stages:
    -   build
    -   test
    -   package

maven-build:
    stage:  build
    image:  maven:3.8.1-jdk-17
    script:
        - git clone 'git_repo_url'
        - cd 'folder_path'
        - mvn clean compile

maven-test:    
    stage:  test
    image:  maven:3.8.1-jdk-17
    script:
        - cd 'folder_path'
        - mvn test

maven-package:
    stage:  package
    image:  maven:3.8.1-jdk-17
    script:
        - cd 'folder_path'
        - mvn package
