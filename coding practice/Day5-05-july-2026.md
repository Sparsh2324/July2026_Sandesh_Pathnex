##  ansible task - Handler to restart nginx
---

-   name:   Manage config file
    hosts:  all
    become: yes

    tasks:
        -   name:   Copy config file
            copy:
                src:    index.html
                dest:   /var/www/html/index.html
            notify: restart Nginx service

    handlers:
        -   name:   restart nginx service
            service:
                name:   nginx
                state:  restarted

## Terraform task   -   EC2 with user data (c6i.12xlarge)

provider "aws" {
    region = "us-east-1"
}

resource "aws-instance" "pathnexEC2" {
    ami =   "<ami_id>"
    instance_type   =   "c6i.12xlarge"

    user_data   =   <<EOF

    #!/bin/bash
    yum install -y httpd
    systemctl start httpd
    EOF

    tags    =   {
        Name    =   "pathnexEC2"
    }
}

## Kubernetes Task - limit the resource

    name:   pathnex-limited-pod

spec:
    containers:

            image:  nginx
            resources:

## Jenkins Task - pipeline with ENVIRONMENT Variable

pipeline {
    agent any

    environemnt {
        INSTITUTION_NAME    =   "Pathnex"
    }

    stages {
        stage ('print') {
            steps {
                sh 'echo "Welcome to $INSTITUTION_NAME Devops Training"
            }
        }

        stage ('maven build') {
            steps {
                sh 'mvn clean package -Dinstitution.name=INSTITUTION_NAME'
            }
        }
    }
}
